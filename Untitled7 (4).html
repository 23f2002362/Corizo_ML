import matplotlib.pyplot as plt
import seaborn as sns
from sklearn.cluster import KMeans
from sklearn.preprocessing import StandardScaler
while True:
    print("Options:")
    print("1. Show a chart of height and weight based on gender")
    print("2. Show a chart of healthy vs. unhealthy people")
    print("3. Show a chart of active vs. non-active people")
    print("4. Show a chart of age categories")
    print("5. Exit")
    choice = input("Enter your choice (1/2/3/4/5): ")

    if choice == '1':
        # Chart of height and weight based on gender
        sns.scatterplot(x='height', y='weight', hue='gender', data=pro)
        plt.title("Height and Weight by Gender")
        plt.xlabel("Height")
        plt.ylabel("Weight")
        plt.show()

    elif choice == '2':
        # Chart of healthy vs. unhealthy people based on cholesterol
        sns.countplot(x='cholesterol', data=pro, hue='cardio')
        plt.title("Healthy vs. Unhealthy People")
        plt.xlabel("Cholesterol")
        plt.ylabel("Count")
        plt.show()

    elif choice == '3':
        # Chart of active vs. non-active people
        sns.countplot(x='active', data=pro)
        plt.title("Active vs. Non-Active People")
        plt.xlabel("Active")
        plt.ylabel("Count")
        plt.show()

    elif choice == '4':
        # Chart of age categories
        pro['age_years'] = pro['age'] / 365
        age_bins = [18, 30, 40, 50, 60, 100]
        age_labels = ['18-30', '31-40', '41-50', '51-60', '61+']
        pro['age_category'] = pd.cut(pro['age_years'], bins=age_bins, labels=age_labels)
        sns.countplot(x='age_category', data=pro)
        plt.title("Age Categories")
        plt.xlabel("Age Category")
        plt.ylabel("Count")
        plt.show()

    elif choice == '5':
        break
    else:
        print("Invalid choice. Please choose a valid option.")
